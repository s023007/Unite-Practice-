حل أسئلة الوحدة الأولى

السؤال (١): في الشكل المقابل

من الرسم نلاحظ أن المنحنى قطع مكافئ مفتوح إلى أعلى، ورأسه تقريبًا عند النقطة (٢ ، −٣).

١) نقطة التحوّل (رأس المنحنى) هي: (٢ ، −٣) (رأس المنحنى).

٢) معادلة محور التماثل: يمر محور التماثل بالرأس ويوازي محور الصادات، إذًا س = ٢.

٣) القيمة الصغرى للدالة: لأن المنحنى مفتوح إلى أعلى، فإن أصغر قيمة لـ ص عند رأس المنحنى، أي ص = −٣.

السؤال (٢): للدالة ص = ٢ − (س + ٣)٢

الدالة على الصورة: ص = ج − (س − ب)٢ ⇒ منحنى مفتوح إلى أسفل ⇒ له قيمة عظمى.

رأس المنحنى عند (−٣ ، ٢) (رأس المنحنى)، إذًا القيمة العظمى هي ٢.

السؤال (٣): للدالة من الشكل ص = (س ± ٢)٢ + ٣

الصورة العامة: ص = (س − ب)٢ + ج ⇒ منحنى مفتوح إلى أعلى ⇒ له قيمة صغرى.

قيمة صغرى = ج = ٣.

السؤال (٤): للدالة ص = س٢ − ٤س + ١

أ = ١ ، ب = −٤ ، ج = ١. نحسب س عند رأس المنحنى من −ب ٢أ :

سرأس = −(−٤) ٢ × ١ = ٤ ٢ = ٢

نعوّض في الدالة: ص = ٢٢ − ٤×٢ + ١ = ٤ − ٨ + ١ = −٣

إذًا نقطة التحوّل (رأس المنحنى) هي (٢ ، −٣) (رأس المنحنى).

السؤال (٥): في الشكل المقابل

المنحنى قطع مكافئ مفتوح إلى أسفل، ورأسه عند النقطة تقريبًا (١ ، ٢).

١) نقطة التحوّل (رأس المنحنى): (١ ، ٢) (رأس المنحنى).

٢) محور التماثل: س = ١.

٣) القيمة العظمى للدالة: ص = ٢.

السؤال (٦): ص = س٢ + ٤س + ١ (إكمال المربع بطريقتين)

الطريقة الأولى: إكمال المربع مباشرة

س٢ + ٤س + ١ = (س٢ + ٤س + ٤) − ٤ + ١ = (س + ٢)٢ − ٣

إذًا الصورة على شكل مربع كامل: ص = (س + ٢)٢ − ٣.

الطريقة الثانية: باستخدام رأس المنحنى

أ = ١ ، ب = ٤ ⇒ سرأس = −٤ ٢ × ١ = −٤ ٢ = −٢

نعوّض في الدالة: ص = (−٢)٢ + ٤(−٢) + ١ = ٤ − ٨ + ١ = −٣

إذًا نقطة التحوّل (رأس المنحنى) هي (−٢ ، −٣) (رأس المنحنى)، وصورة الدالة: ص = (س + ٢)٢ − ٣.

السؤال (٨): ص = س٢ − ٦س + ٥ (إكمال المربع بطريقتين)

الطريقة الأولى: إكمال المربع مباشرة

س٢ − ٦س + ٥ = (س٢ − ٦س + ٩) − ٩ + ٥ = (س − ٣)٢ − ٤

إذًا: ص = (س − ٣)٢ − ٤.

الطريقة الثانية: باستخدام رأس المنحنى

أ = ١ ، ب = −٦ ⇒ سرأس = −(−٦) ٢ × ١ = ٦ ٢ = ٣

نعوّض في الدالة: ص = ٣٢ − ٦×٣ + ٥ = ٩ − ١٨ + ٥ = −٤

إذًا نقطة التحوّل (رأس المنحنى) هي (٣ ، −٤) (رأس المنحنى)، وصورة الدالة: ص = (س − ٣)٢ − ٤.

السؤال (٩): نقطة تقاطع ص = س٢ − ٤س + ١ مع محور الصادات

على محور الصادات يكون س = ٠.

ص = ٠٢ − ٤×٠ + ١ = ١

إذًا نقطة التقاطع هي (٠ ، ١).

السؤال (١٠): رأس منحنى الدالة ص = (س − أ)٢ + ب هو (٢ ، −٥)

رأس المنحنى للدالة على هذه الصورة هو (أ ، ب).

بما أن الرأس = (٢ ، −٥)، إذًا أ = ٢ ، ب = −٥.

أ + ب = ٢ + (−٥) = −٣.

إذن أ + ب = −٣.

السؤال (١١): نوع جذري المعادلة س٢ − ٣س − ١٠ = ٠

أ = ١ ، ب = −٣ ، ج = −١٠.

المميز Δ = ب٢ − ٤أج = (−٣)٢ − ٤×١×(−١٠) = ٩ + ٤٠ = ٤٩ > ٠

إذًا للمعادلة جذران حقيقيان مختلفان.

السؤال (١٢): عدد جذور المعادلة س٢ − ٦س + ٩ = ٠

Δ = (−٦)٢ − ٤×١×٩ = ٣٦ − ٣٦ = ٠

إذًا لها جذر حقيقي واحد مكرر.

السؤال (١٣): إذا كان ب٢ − ٤أج < ٠

إذا كان Δ < ٠ فإن المعادلة لا تملك جذورًا حقيقية.

السؤال (١٤): س٢ − ك س + ٩ = ٠ لها جذران متساويان

شرط الجذرين المتساويين: Δ = ٠.

Δ = ك٢ − ٤×١×٩ = ك٢ − ٣٦ = ٠

ك٢ = ٣٦ ⇒ ك = ±٦.

إذن ك = ±٦.

السؤال (١٥): النظام ص = س + ١ ، س٢ + ص٢ = ١٣

نعوّض ص = س + ١ في المعادلة الثانية:

س٢ + (س + ١)٢ = ١٣ ⇒ س٢ + س٢ + ٢س + ١ = ١٣ ⇒ ٢س٢ + ٢س − ١٢ = ٠

نقسم على ٢: س٢ + س − ٦ = ٠ ⇒ (س + ٣)(س − ٢) = ٠ ⇒ س = ٢ أو س = −٣

عند س = ٢ ⇒ ص = ٣ عند س = −٣ ⇒ ص = −٢

الحلان: (٢ ، ٣) و (−٣ ، −٢)، والزوج المطلوب غالبًا (٢ ، ٣).

السؤال (١٦): النظام ص = س − ١ ، س٢ + ص٢ = ٥

ص = س − ١ ⇒ نعوّض:

س٢ + (س − ١)٢ = ٥ ⇒ س٢ + س٢ − ٢س + ١ = ٥ ⇒ ٢س٢ − ٢س − ٤ = ٠

نقسم على ٢: س٢ − س − ٢ = ٠ ⇒ (س − ٢)(س + ١) = ٠ ⇒ س = ٢ أو س = −١

عند س = ٢ ⇒ ص = ١ عند س = −١ ⇒ ص = −٢

الحلان: (٢ ، ١) و (−١ ، −٢)، والزوج المطلوب غالبًا (٢ ، ١).

السؤال (١٧): إذا كانت (٢ ، ١) حلًا للنظام ص = س − أ ، س٢ + ص٢ = ب

من ص = س − أ وبالتعويض: ١ = ٢ − أ ⇒ أ = ١

ومن س٢ + ص٢ = ب: ٢٢ + ١٢ = ٤ + ١ = ٥ ⇒ ب = ٥

أ + ب = ١ + ٥ = ٦

إذن أ + ب = ٦.

السؤال (١٨): إذا كانت (٢ ، ٣) حلًا للنظام ص = س − أ ، ص٢ − س٢ = ب

من ص = س − أ: ٣ = ٢ − أ ⇒ أ = −١

من ص٢ − س٢ = ب: ٣٢ − ٢٢ = ٩ − ٤ = ٥ ⇒ ب = ٥

أ + ب = −١ + ٥ = ٤

إذن أ + ب = ٤.

السؤال (١٩): عدد نقاط تقاطع ص = ٣س + ٢ مع ص = س٢

عند التقاطع: س٢ = ٣س + ٢ ⇒ س٢ − ٣س − ٢ = ٠

Δ = (−٣)٢ − ٤×١×(−٢) = ٩ + ٨ = ١٧ > ٠

إذًا يوجد حلّان حقيقيان ⇒ يتقاطعان في نقطتين.

السؤال (٢٠): اختيار الدالة ذات جذور حقيقية مختلفة (ملاحظات عامة)

لأن نص الدوال في الملف غير واضح، نذكر القاعدة العامة للطلاب:

  • إذا كانت الدالة على الصورة أ س٢ + ب س + ج = ٠، نحسب Δ = ب٢ − ٤أج.
  • إذا كان Δ > ٠ ⇒ جذران حقيقيان مختلفان.
  • إذا كان Δ = ٠ ⇒ جذران حقيقيان متساويان.
  • إذا كان Δ < ٠ ⇒ لا توجد جذور حقيقية.
السؤال (٢١): واحدة من هذه الدوال لها جذران متساويان

ننظر إلى موضع منحنى كل دالة بالنسبة لمحور السينات:

  • يقطع المحور في نقطتين ⇒ جذران مختلفان.
  • لا يقطع المحور ⇒ لا جذور حقيقية.
  • يلامس المحور في نقطة واحدة فقط (رأس المنحنى على المحور) ⇒ جذران حقيقيان متساويان.

من الرسوم الأربعة، المنحنى الأخضر في أقصى اليمين يلامس محور السينات عند نقطة واحدة فقط، لذلك جذراه متساويان.

إذن الرسم الصحيح هو الرسم الرابع (الأخضر في أقصى اليمين).

السؤال (٢٢): التمثيل البياني المقابل هو بيان الدالة ص = س٢ − ٤س + ٣

نكتب الدالة على صورة إكمال المربع:

س٢ − ٤س + ٣ = (س٢ − ٤س + ٤) − ٤ + ٣ = (س − ٢)٢ − ١

إذًا نقطة التحوّل (رأس المنحنى) هي (٢ ، −١) (رأس المنحنى)، ومحور التماثل س = ٢.

بحل المعادلة س٢ − ٤س + ٣ = ٠ نحصل على الجذرين ١ ، ٣، والتقاطع مع محور الصادات هو (٠ ، ٣).

  • (١) نقطة التحوّل (−١ ، ٢) ⟵ خطأ.
  • (٢) معادلة محور التماثل س = ٢ ⟵ صح.
  • (٣) جذرا المعادلة ص = ٠ هما (١ ، ٠) و(٣ ، ٠) ⟵ صح.
  • (٤) نقطة التقاطع مع الصادات (٠ ، ٣) ⟵ صح.
السؤال (٢٣): عدد حلول المعادلة ٢س٢ − ٥س + ٢ = ٠

أ = ٢ ، ب = −٥ ، ج = ٢.

Δ = (−٥)٢ − ٤×٢×٢ = ٢٥ − ١٦ = ٩ > ٠

إذًا يوجد حلّان حقيقيان.

السؤال (٢٤): عدد حلول المعادلة ٤س٢ − ١٢س + ٩ = ٠

أ = ٤ ، ب = −١٢ ، ج = ٩.

Δ = (−١٢)٢ − ٤×٤×٩ = ١٤٤ − ١٤٤ = ٠

إذًا لها حل حقيقي واحد مكرر.

السؤال (٢٥): إذا كان مميز المعادلة ٤س٢ − ٦س + ك = ٠ يساوي ٨، أوجد ك

Δ = ب٢ − ٤أج = (−٦)٢ − ٤×٤×ك = ٣٦ − ١٦ك

نحلي: ٣٦ − ١٦ك = ٨ ⇒ ١٦ك = ٢٨

ك = ٢٨ ١٦ = ٧ ٤

إذن ك = ٧/٤ (كسر في أبسط صورة).

السؤال (٢٦): إذا كان مميز المعادلة ك س٢ − ٦س + ١ = ٠ يساوي ٤، أوجد ك

Δ = (−٦)٢ − ٤ك×١ = ٣٦ − ٤ك

٣٦ − ٤ك = ٤ ⇒ ٤ك = ٣٢ ⇒ ك = ٨

إذن ك = ٨.

السؤال (٢٧): اكتب ما يلي على صورة مربع كامل

١) س٢ − ٤س + ٥

س٢ − ٤س + ٥ = (س٢ − ٤س + ٤) + ١ = (س − ٢)٢ + ١

٢) س٢ + ٦س + ٣

س٢ + ٦س + ٣ = (س٢ + ٦س + ٩) − ٦ = (س + ٣)٢ − ٦

٣) س٢ − ٥س + ٧

نصف (−٥) هو −٢٫٥، مربّعه ٦٫٢٥:

س٢ − ٥س + ٧ = (س٢ − ٥س + ٦٫٢٥) + ٠٫٧٥ = (س − ٢٫٥)٢ + ٠٫٧٥

السؤال (٢٨): ص = س٢ − ٢س − ٣

نكمل المربع: س٢ − ٢س − ٣ = (س٢ − ٢س + ١) − ١ − ٣ = (س − ١)٢ − ٤

  • نقطة التحوّل (رأس المنحنى): (١ ، −٤).
  • محور التماثل: س = ١.
  • القيمة الصغرى: ص = −٤.
  • التقاطع مع محور السينات: بحل س٢ − ٢س − ٣ = ٠ ⇒ (س − ٣)(س + ١) = ٠ ⇒ س = ٣ ، −١ ⇒ النقطتان (٣ ، ٠) ، (−١ ، ٠).
  • التقاطع مع محور الصادات: عند س = ٠ ⇒ ص = −٣ ⇒ (٠ ، −٣).
السؤال (٢٩): حدد نوع الجذور

١) س٢ − ٦س + ٩ = ٠

Δ = (−٦)٢ − ٤×١×٩ = ٣٦ − ٣٦ = ٠ ⇒ جذران حقيقيان متساويان.

٢) س٢ + ٥س + ٦ = ٠

Δ = ٥٢ − ٤×١×٦ = ٢٥ − ٢٤ = ١ > ٠ ⇒ جذران حقيقيان مختلفان.

٣) س٢ − ٣س + ٥ = ٠

Δ = (−٣)٢ − ٤×١×٥ = ٩ − ٢٠ = −١١ < ٠ ⇒ لا جذور حقيقية (جذران عقديان).

السؤال (٣٠): أوجد ك بحيث يكون جذرا س٢ − ك س + ٦ = ٠ متساويين

Δ = ك٢ − ٤×١×٦ = ك٢ − ٢٤ = ٠

ك٢ = ٢٤ ⇒ ك = ± √٢٤ = ± ٢√٦

إذن ك = ±٢√٦.

السؤال (٣١): أوجد ك بحيث يكون جذرا س٢ − ٨س + ك = ٠ متساويين

Δ = (−٨)٢ − ٤×١×ك = ٦٤ − ٤ك = ٠

٤ك = ٦٤ ⇒ ك = ١٦

إذن ك = ١٦.

السؤال (٣٢): حل المتباينة س٢ − ٣س − ١٠ < ٠

نحل المعادلة س٢ − ٣س − ١٠ = ٠ ⇒ (س − ٥)(س + ٢) = ٠ ⇒ س = ٥ ، −٢

المنحنى مفتوح إلى أعلى، فيكون أقل من صفر بين الجذرين: −٢ < س < ٥.

السؤال (٣٣): حل المتباينة س٢ − س − ١٢ ≥ ٠

نحل المعادلة س٢ − س − ١٢ = ٠ ⇒ (س − ٤)(س + ٣) = ٠ ⇒ س = ٤ ، −٣

المنحنى مفتوح إلى أعلى، فيكون أكبر أو يساوي صفر خارج الفترة: س ≤ −٣ أو س ≥ ٤.

السؤال (٣٤): حل المتباينة س٢ − ٣س < ٠

س٢ − ٣س = س(س − ٣) ⇒ الجذور: ٠ ، ٣.

المنحنى مفتوح إلى أعلى ⇒ أقل من صفر بين الجذرين: ٠ < س < ٣.

السؤال (٣٥): حل المتباينة س٢ − ٦س + ٩ ≥ ٠

س٢ − ٦س + ٩ = (س − ٣)٢ ≥ ٠ لجميع قيم س.

مجموعة الحل: جميع الأعداد الحقيقية.

السؤال (٣٦): حل المتباينة س٢ < ٣س − ٥

ننقل الأطراف: س٢ − ٣س + ٥ < ٠

Δ = (−٣)٢ − ٤×١×٥ = ٩ − ٢٠ = −١١ < ٠

المنحنى مفتوح إلى أعلى ولا يقطع محور السينات، لذلك قيمته دائمًا أكبر من صفر ⇒ لا يوجد س يجعلها أقل من صفر.

مجموعة الحل: ∅ (لا حل).

السؤال (٣٧): حل المتباينة س٢ ≤ ١٢س − ٣٦

س٢ − ١٢س + ٣٦ ≤ ٠ ، والحدّ الأيسر يساوي (س − ٦)٢.

(س − ٦)٢ ≤ ٠ يحدث فقط عندما س − ٦ = ٠ ⇒ س = ٦.

مجموعة الحل: {٦}.

السؤال (٣٨): حل المعادلات الآتية باستخدام إكمال المربع

١) س٢ + ٩س − ١٠ = ٠

س٢ + ٩س = ١٠

نصف ٩ هو ٤٫٥، مربّعه ٢٠٫٢٥:

س٢ + ٩س + ٢٠٫٢٥ = ١٠ + ٢٠٫٢٥ ⇒ (س + ٤٫٥)٢ = ٣٠٫٢٥

س + ٤٫٥ = ± √٣٠٫٢٥ = ±٥٫٥ ⇒ س = ١ أو س = −١٠

٢) س٢ − ٥س + ٦ = ٠

يمكن باستخدام التحليل: (س − ٢)(س − ٣) = ٠ ⇒ س = ٢ أو س = ٣.

السؤال (٣٩): حل المعادلتين بالصيغة العامة

١) ٢س٢ + ٥س − ٦ = ٠

أ = ٢ ، ب = ٥ ، ج = −٦

س = −٥ ± √(٥٢ − ٤×٢×(−٦)) ٢×٢ = −٥ ± √٧٣ ٤

٢) س٢ + ٣س − ٥ = ٠

أ = ١ ، ب = ٣ ، ج = −٥

س = −٣ ± √(٣٢ − ٤×١×(−٥)) ٢×١ = −٣ ± √٢٩ ٢

السؤال (٤٠): أوجد المميز ثم نوع الجذور

١) س٢ − ٧س + ٨ = ٠

Δ = (−٧)٢ − ٤×١×٨ = ٤٩ − ٣٢ = ١٧ > ٠ ⇒ جذران حقيقيان مختلفان.

٢) ٣س٢ + ١٠س − ٢ = ٠

Δ = ١٠٢ − ٤×٣×(−٢) = ١٠٠ + ٢٤ = ١٢٤ > ٠ ⇒ جذران حقيقيان مختلفان.

السؤال (٤١): استخدام المميز لتحديد عدد الجذور

بسبب عدم وضوح أرقام المعادلة في الملف، نذكر القاعدة العامة:

  • للمعادلة أ س٢ + ب س + ج = ٠ نحسب المميز Δ = ب٢ − ٤أج.
  • إذا كان Δ > ٠ ⇒ جذران حقيقيان مختلفان.
  • إذا كان Δ = ٠ ⇒ جذر حقيقي واحد مكرر.
  • إذا كان Δ < ٠ ⇒ لا جذور حقيقية.
السؤال (٤٣): أوجد قيم س الحقيقية للمعادلات الآتية

١) س٤ − ١٣س٢ + ٣٦ = ٠

نضع م = س٢ ⇒ م٢ − ١٣م + ٣٦ = ٠ ⇒ (م − ٤)(م − ٩) = ٠ ⇒ م = ٤ أو م = ٩

س٢ = ٤ ⇒ س = ±٢ س٢ = ٩ ⇒ س = ±٣

الحلول: −٣ ، −٢ ، ٢ ، ٣.
